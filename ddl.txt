-- 1. 학생 정보 테이블 (회원 계정 및 보유 자산)
CREATE TABLE students
(
    id            INT AUTO_INCREMENT PRIMARY KEY,
    student_id    VARCHAR(100) NOT NULL UNIQUE,
    password      VARCHAR(100) NOT NULL,
    name          VARCHAR(50)  NOT NULL,
    grade         INT          NOT NULL,
    class_name    VARCHAR(50)  NOT NULL,
    class_number  INT          NOT NULL,
    register_year INT          NOT NULL,
    total_coupon  INT DEFAULT 0,
    total_point   INT DEFAULT 0,
    created_date  DATETIME     NOT NULL,
    updated_date  DATETIME
);

-- 2. 상점 쿠폰 기본 정보 테이블
CREATE TABLE coupons
(
    coupon_id    INT AUTO_INCREMENT PRIMARY KEY,
    student_id   VARCHAR(100) NOT NULL,
    name         VARCHAR(100) NOT NULL,
    price        INT          NOT NULL,
    created_date DATETIME     NOT NULL,
    FOREIGN KEY (student_id) REFERENCES students (student_id) ON DELETE CASCADE
);

-- 3. 학생들의 쿠폰 구매 내역 테이블
CREATE TABLE coupon_purchase
(
    purchase_id  INT AUTO_INCREMENT PRIMARY KEY,
    price        INT          NOT NULL,
    name         VARCHAR(100) NOT NULL,
    state        VARCHAR(100) NOT NULL,
    created_date DATETIME     NOT NULL,
    updated_date DATETIME     NULL,
    coupon_id    INT          NOT NULL,
    student_id   VARCHAR(100) NOT NULL,
    FOREIGN KEY (coupon_id) REFERENCES coupons (coupon_id) ON DELETE CASCADE,
    FOREIGN KEY (student_id) REFERENCES students (student_id) ON DELETE CASCADE
);

-- 4. 포인트 획득/지급 이력 테이블
CREATE TABLE get_points
(
    get_point_id INT AUTO_INCREMENT PRIMARY KEY,
    student_id   VARCHAR(100) NOT NULL,
    content      VARCHAR(300),
    point        INT,
    created_date DATETIME     NOT NULL,
    FOREIGN KEY (student_id) REFERENCES students (student_id) ON DELETE CASCADE
);

-- 5. 투자 심리 자극용 독립 뉴스 테이블
CREATE TABLE news
(
    news_id      INT AUTO_INCREMENT PRIMARY KEY,
    content      VARCHAR(300),
    created_date DATETIME NOT NULL
);

-- 6. 주식 종목 기본 정보 테이블 (orders보다 먼저 생성되어야 함)
CREATE TABLE stocks
(
    stock_id            INT AUTO_INCREMENT PRIMARY KEY,
    name                VARCHAR(100) NOT NULL,
    content             VARCHAR(100),
    publication_balance INT,
    publication_price   INT          NOT NULL,
    prev_price          INT          NULL,
    created_date        DATETIME     NOT NULL
);

-- 7. 주식 주문 요청 내역 테이블 (매수/매도 대기 물량 포함)
CREATE TABLE orders
(
    order_id     INT AUTO_INCREMENT PRIMARY KEY,
    content      VARCHAR(300),
    price        INT,
    amount       INT,
    state        VARCHAR(100),
    created_date DATETIME     NOT NULL,
    updated_date DATETIME,
    deleted_date DATETIME,
    student_id   VARCHAR(100) NOT NULL,
    stock_id     INT          NOT NULL,
    FOREIGN KEY (student_id) REFERENCES students (student_id) ON DELETE CASCADE,
    FOREIGN KEY (stock_id) REFERENCES stocks (stock_id) ON DELETE CASCADE
);

-- 8. 최종 매칭 엔진에 의해 성사된 주식 거래(체결) 내역 테이블
CREATE TABLE transactions
(
    transaction_id INT AUTO_INCREMENT PRIMARY KEY,
    buy_order_id   INT      NOT NULL,
    sell_order_id  INT      NULL,
    created_date   DATETIME NOT NULL,
    FOREIGN KEY (buy_order_id) REFERENCES orders (order_id) ON DELETE CASCADE,
    FOREIGN KEY (sell_order_id) REFERENCES orders (order_id) ON DELETE SET NULL
);
