-- 1. 학생 정보 테이블 (회원 계정 및 보유 자산)
CREATE TABLE students
(
    id           INT AUTO_INCREMENT PRIMARY KEY,
    studentId    VARCHAR(100) NOT NULL UNIQUE,
    password     VARCHAR(100) NOT NULL,
    name         VARCHAR(50)  NOT NULL,
    grade        INT          NOT NULL,
    className    VARCHAR(50)  NOT NULL,
    registerYear INT          NOT NULL,
    totalCoupon  INT    DEFAULT 0,
    totalPoint   INT DEFAULT 0,
    createdDate  DATETIME     NOT NULL,
    updatedDate  DATETIME
);

-- 2. 상점 쿠폰 기본 정보 테이블
CREATE TABLE coupons
(
    couponId    INT AUTO_INCREMENT PRIMARY KEY,
    studentId   VARCHAR(100) NOT NULL,
    name        VARCHAR(100) NOT NULL,
    price       INT          NOT NULL,
    createdDate DATETIME     NOT NULL,
    FOREIGN KEY (studentId) REFERENCES students (studentId) ON DELETE CASCADE
);

-- 3. 학생들의 쿠폰 구매 내역 테이블
CREATE TABLE couponPurchase
(
    purchaseId  INT AUTO_INCREMENT PRIMARY KEY,
    price       INT          NOT NULL,
    name        VARCHAR(100) NOT NULL,
    state       VARCHAR(100) NOT NULL,
    createdDate DATETIME     NOT NULL,
    updatedDate DATETIME     NULL,
    couponId    INT          NOT NULL,
    studentId   INT          NOT NULL,                                 -- students.id 매칭용 INT 타입
    FOREIGN KEY (couponId) REFERENCES coupons (couponId) ON DELETE CASCADE,
    FOREIGN KEY (studentId) REFERENCES students (id) ON DELETE CASCADE -- coupons 대신 students 참조로 정정
);

-- 4. 포인트 획득/지급 이력 테이블
CREATE TABLE getPoints
(
    getPointId  INT AUTO_INCREMENT PRIMARY KEY,
    studentId   INT      NOT NULL,                                     -- students.id 매칭용 INT 타입
    content     VARCHAR(300),
    point       INT,
    createdDate DATETIME NOT NULL,
    FOREIGN KEY (studentId) REFERENCES students (id) ON DELETE CASCADE -- 고유 id 참조로 데이터 무결성 확보
);

-- 5. 투자 심리 자극용 독립 뉴스 테이블
CREATE TABLE news
(
    newsId      INT AUTO_INCREMENT PRIMARY KEY,
    content     VARCHAR(300),
    createdDate DATETIME NOT NULL
);

-- 6. 주식 종목 기본 정보 테이블 (orders보다 먼저 생성되어야 함)
CREATE TABLE stocks
(
    stockId            INT AUTO_INCREMENT PRIMARY KEY,
    name               VARCHAR(100) NOT NULL,
    content            VARCHAR(100),
    publicationBalance INT,
    publicationPrice   INT          NOT NULL,
    prevPrice          INT          NULL,
    createdDate        DATETIME     NOT NULL
);
DROP TABLE IF EXISTS students;

-- 7. 주식 주문 요청 내역 테이블 (매수/매도 대기 물량 포함)
CREATE TABLE orders
(
    orderId     INT AUTO_INCREMENT PRIMARY KEY,
    content     VARCHAR(300),                                           -- '매수' 또는 '매도' 구분
    price       INT,
    amount      INT,
    state       VARCHAR(100),                                           -- '대기', '체결', '취소' 상태 관리
    createdDate DATETIME NOT NULL,
    updatedDate DATETIME,
    deletedDate DATETIME,
    studentId   INT      NOT NULL,                                      -- students.id 매칭용 INT 타입
    stockId     INT      NOT NULL,
    FOREIGN KEY (studentId) REFERENCES students (id) ON DELETE CASCADE, -- 고유 id 참조로 정정
    FOREIGN KEY (stockId) REFERENCES stocks (stockId) ON DELETE CASCADE
);

-- 8. 최종 매칭 엔진에 의해 성사된 주식 거래(체결) 내역 테이블 (완성본)
CREATE TABLE transactions
(
    transactionId INT AUTO_INCREMENT PRIMARY KEY,
    buyOrderId    INT      NOT NULL, -- 매수한 주문요청 일련번호
    sellOrderId   INT      NULL,     -- 매도한 주문요청 일련번호 (시스템 발행 주식 매수 시 NULL 허용 조약 반영)
    createdDate   DATETIME NOT NULL,
    FOREIGN KEY (buyOrderId) REFERENCES orders (orderId) ON DELETE CASCADE,
    FOREIGN KEY (sellOrderId) REFERENCES orders (orderId) ON DELETE SET NULL
);
